\begin{frame}{Not just relative decline: men's wages have \alert{declined} in denser places while women's have increased everywhere}
	\bitem
		\item \alert{\textbf{Men's}} wages have \alert{\textbf{declined}} in denser labor markets since 2000.
		\item \alert{\textbf{Women's wages}} have \alert{\textbf{increased}} everywhere. Increase was faster in denser markets during 1990-1970. The opposite is true for 2000-2020.
	\eitem
	\begin{columns}
		\column{.5\textwidth}
		\addfig{1}{../2_analysis/output/figures/d_wage_deciles_men}
		\column{.5\textwidth}
		\addfig{1}{../2_analysis/output/figures/d_wage_deciles_women}
	\end{columns}
\grnote
\end{frame}
